// Комментарий

// Секция для задания вершин, их идентификаторов и подписей
vertex
{
    { <id> [: <string>]; }*
}

// Секция для определения связей между вершинами и типом графа
graph
{
    ( directed | undirected | ... );
    { <id>: ( <id> | "[" { <id> }+ "]" ) "#" NUMLIT; }*
}

// Секция для выполнения алгоритмов над графом
algorithm
{
    { <id> { <param>, } * ; }
}
